ReGenESyS - REborned GENeric and Expansible SYstem Simulator  v22.01 (wealljudes)
Genesys is a result of teaching and research activities of Professor Dr. Ing Rafael Luiz Cancian. It began in early 2002 as a way to teach students the basics and simulation techniques of systems implemented by other comercial simulation tools, such as Arena. In Genesys development he replicated all the SIMAN language, used by Arena software, and Genesys has become a clone of that tool, including its graphical interface. Genesys allowed the inclusion of new simulation components through dynamic link libraries and also the parallel execution of simulation models in a distributed environment. The development of Genesys continued until 2007, when the professor stopped teaching systems simulation classes. Ten years later the professor starts again to teach systems simulation classes and to carry out scientific research in the area. So in 2018 Genesys is reborn as ReGenesys, with new language and programming techniques, and even more ambitious goals.

75 void Route::_execute(Entity* entity) {
76  Station* destinyStation = _station;
77  if (_routeDestinationType == Route::DestinationType::BySequence) {
78  unsigned int sequenceId = static_cast<unsigned int> (entity->getAttributeValue("Entity.Sequence"));
79  unsigned int step = static_cast<unsigned int> (entity->getAttributeValue("Entity.SequenceStep"));
80  Sequence* sequence = static_cast<Sequence*> (_parentModel->getData()->getData(Util::TypeOf<Sequence>(), sequenceId));
81  SequenceStep* seqStep = sequence->getSteps()->getAtRank(step);
82  if (seqStep == nullptr) {
83  step = 0;
84  seqStep = sequence->getSteps()->getAtRank(step);
85  assert(seqStep != nullptr);
86  }
87  destinyStation = seqStep->getStation();
88  for (SequenceStep::Assignment* assignment : *seqStep->getAssignments()) {
89  _parentModel->parseExpression(assignment->getDestination() + "=" + assignment->getExpression());
90  }
91  entity->setAttributeValue("Entity.SequenceStep", step + 1.0);
92  }
93  if (_reportStatistics)
94  _numberIn->incCountValue();
95  // adds the route time to the TransferTime statistics / attribute related to the Entitys
96  double routeTime = _parentModel->parseExpression(_routeTimeExpression) * Util::TimeUnitConvert(_routeTimeTimeUnit, _parentModel->getSimulation()->getReplicationBaseTimeUnit());
97  if (entity->getEntityType()->isReportStatistics())
98  entity->getEntityType()->addGetStatisticsCollector(entity->getEntityTypeName() + ".TransferTime")->getStatistics()->getCollector()->addValue(routeTime);
99  entity->setAttributeValue("Entity.TotalTransferTime", entity->getAttributeValue("Entity.TotalTransferTime") + routeTime);
100  if (routeTime > 0.0) {
101  // calculates when this Entity will reach the end of this route and schedule this Event
102  double routeEndTime = _parentModel->getSimulation()->getSimulatedTime() + routeTime;
103  Event* newEvent = new Event(routeEndTime, entity, destinyStation->getEnterIntoStationComponent());
104  _parentModel->getFutureEvents()->insert(newEvent);
105  _parentModel->getTracer()->traceSimulation(this, "End of route of "/*entity " + std::to_string(entity->entityNumber())*/ + entity->getName() + " to the component \"" + destinyStation->getEnterIntoStationComponent()->getName() + "\" was scheduled to time " + std::to_string(routeEndTime));
106  } else {
107  // send without delay
108  _parentModel->sendEntityToComponent(entity, destinyStation->getEnterIntoStationComponent(), 0.0);
109  }
110 }

144 bool Route::_check(std::string* errorMessage) {
145  //include attributes needed
146  ModelDataManager* elements = _parentModel->getData();
147  std::vector<std::string> neededNames = {"Entity.TotalTransferTime", "Entity.Station"};
148  std::string neededName;
149  for (unsigned int i = 0; i < neededNames.size(); i++) {
150  neededName = neededNames[i];
151  if (elements->getData(Util::TypeOf<Attribute>(), neededName) == nullptr) {
152  Attribute* attr1 = new Attribute(_parentModel, neededName);
153  elements->insert(attr1);
154  }
155  }
156  // include StatisticsCollector needed in EntityType
157  std::list<ModelData*>* enttypes = elements->getElementList(Util::TypeOf<EntityType>())->list();
158  for (ModelData* modeldatum : *enttypes) {
159  if (modeldatum->isReportStatistics())
160  static_cast<EntityType*> (modeldatum)->addGetStatisticsCollector(modeldatum->getName() + ".TransferTime"); // force create this CStat before simulation starts
161  }
162  bool resultAll = true;
163  resultAll &= _parentModel->checkExpression(_routeTimeExpression, "Route time expression", errorMessage);
164  if (this->_routeDestinationType == Route::DestinationType::Station) {
165  resultAll &= _parentModel->getData()->check(Util::TypeOf<Station>(), _station, "Station", errorMessage);
166  if (resultAll) {
167  resultAll &= _station->getEnterIntoStationComponent() != nullptr;
168  if (!resultAll) {
169  errorMessage->append("Station has no component to enter into it");
170  }
171  }
172  }
173  //_model->getParent()->getPluginManager()->
174  return resultAll;
175 }
